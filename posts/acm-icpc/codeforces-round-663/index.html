int cost(int col, int mask){
    int res = 0;
    for(int i = 1; i <= n; i++) {
        int x = s[i][col] - '0';
        int y = mask & 1;
        res += x != y;
        mask >>= 1;
    }

}
bool check(int x, int y) {
    int A[2][10];
    for(int i = 1; i <= n; i++) {
        int a = x & 1;
        int b = y & 1;
        A[0][i] = a;
        A[1][i] = b;
        x >>= 1;
        y >>= 1;
    }
    int res = A[0][1] ^ A[0][2] ^ A[1][1] ^ A[1][2];

    return res;
}

void dfs(int u, int parent, int d) {
