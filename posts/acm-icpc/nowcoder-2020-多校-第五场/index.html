int n;
int a[MAX];
int b[MAX];
int solve(){
    mset(b, 0);
    int ans = 0;
    for(int i = 1; i <= n; i++) {
        int idx = lower_bound(b, b+ans, a[i]) - b;
        b[idx] = a[i];
        if(idx == ans) ans ++;
    }
    // cout << n - ans << "\n";
    return ans;
}

from math import gcd

def lcm(a:int, b:int) ->int:
    return a * b // gcd(a, b)

n = int(input())
a = [0] + list(map(int, input().split()))
vis = [False for i in range(len(a))]

ans = 1
for i in range(1, n+1):
    x = a[i]
    if vis[x] : continue
    cur = 0
    while not vis[x] :
        vis[x] = True
        cur += 1
        x = a[x]
    ans = lcm(ans, cur)
print(ans)
